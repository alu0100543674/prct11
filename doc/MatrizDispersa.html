def DispersaDensa() 
     mDensa = MatrizDensa.new(@filas, @columnas)
   m = Array.new()
   for i in 0...@filas
             aux = Array.new()
      for j in 0...@columnas
         if (@matriz.include?("#{i},#{j}"))
            aux << @matriz["#{i},#{j}"]
         else
            aux << 0
         end 
      end 
      m << aux
   end 
   mDensa.copia(m)
   mDensa
